PDF Template Playground

Schema 
PdfData {
  title?: string
  description?: string
  submittedAt?: string
  sections?: PdfSection[]
  files?: PdfFile[]
}

PdfSection {
  title?: string
  children: (PdfField | PdfSection)[]
}

PdfField {
  type: 'text' | 'email' | 'phone' | 'number' | 'checkbox' | 'date'
       | 'duration' | 'dropdown' | 'choice' | 'rating' | 'slider'
       | 'media' | 'avatar' | 'signature' | 'location' | 'array'
  name: string
  label?: string
  value: any
  displayValue?: string              // dropdown display label
  mode?: 'date' | 'time' | 'datetime'
  maximum?: number                   // rating, slider
  minimum?: number                   // slider
  options?: {value, label}[]         // choice fields
  fields?: FieldDescriptor[]         // array sub-field descriptors (no value)
  maxColumns?: number                // array: table→card threshold (default 6)
}

FieldDescriptor {
  type: (same as PdfField.type)
  name: string
  label?: string
  mode?: 'date' | 'time' | 'datetime'
  maximum?: number
  minimum?: number
  options?: {value, label}[]
}

PdfFile {
  fileName: string
  fieldName?: string
  thumbnail?: string                 // base64
  localPath?: string
}
